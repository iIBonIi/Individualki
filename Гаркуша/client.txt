#include <iostream>
#include <cstring>
#include <arpa/inet.h>

int main() {
    int sock = 0;
    struct sockaddr_in serv_addr;
    double imt;

    // Создаем сокет
    if ((sock = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
        std::cout << "Ошибка создания сокета" << std::endl;
        return -1;
    }

    // Задаем адрес сервера
    serv_addr.sin_family = AF_INET;
    serv_addr.sin_port = htons(8080); // Порт, который будет использовать сервер

    // Преобразуем адрес IPv4 и IPv6 из текстового формата в бинарный
    if (inet_pton(AF_INET, "127.0.0.1", &serv_addr.sin_addr) <= 0) {
        std::cout << "Неверный адрес" << std::endl;
        return -1;
    }

    // Подключаемся к серверу
    if (connect(sock, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {
        std::cout << "Ошибка подключения" << std::endl;
        return -1;
    }

    // Запрашиваем ИМТ у пользователя
    std::cout << "Введите ваш ИМТ: ";
    std::cin >> imt;

    // Отправляем ИМТ на сервер
    send(sock, &imt, sizeof(imt), 0);

    // Получаем ответ от сервера
    char buffer[1024] = {0};
    recv(sock, buffer, sizeof(buffer), 0);
    std::cout << "Ответ от сервера: " << buffer << std::endl;

    // Закрываем сокет
    close(sock);
    return 0;
}
