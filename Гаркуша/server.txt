#include <iostream>
#include <fstream>
#include <string>
#include <thread>
#include <chrono>

void checkImt(double imt, const std::string& responseFile) {
    std::string response;
    
    // Проверка ИМТ и формирование ответа
    if (imt < 18.5) {
        response = "Недостаточный вес";
    } else if (imt >= 18.5 && imt < 24.9) {
        response = "Норма";
    } else {
        response = "Избыточный вес";
    }

    // Запись ответа в файл
    std::ofstream ofs(responseFile);
    if (ofs.is_open()) {
        ofs << response;
        ofs.close();
    } else {
        std::cout << "Ошибка при открытии файла для записи ответа." << std::endl;
    }
}

int main() {
    double imt;

    // Чтение ИМТ из файла
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1)); // Простая задержка перед проверкой

        std::ifstream ifs("imt.txt");
        if (ifs.is_open()) {
            ifs >> imt;
            ifs.close();
            std::cout << "Получен ИМТ: " << imt << std::endl;
            checkImt(imt, "response.txt"); // Проверка ИМТ и отправка ответа
            break; // Выходим из цикла после получения ИМТ
        }
    }
}
